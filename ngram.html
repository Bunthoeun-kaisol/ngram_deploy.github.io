Part 1: Indexing Process

Implement a basic indexing process aligned with the key concepts of search engine architecture. Below is a function that takes a list of documents (text strings) and creates an inverted index. The inverted index maps terms to the list of document IDs.

Documents:

{
    "Article: Pandas Basics": "This article covers the basics of using Pandas in Python.",
    "Tutorial: Data Visualization": "Learn about data visualization techniques with Python libraries.",
    "Case Study: Sales Analysis": "Analyze sales data using Python for a real-world case study."
}

Code:

function createInvertedIndex(documents) {
    const invertedIndex = {};
    Object.entries(documents).forEach(([docID, text]) => {
        const words = text.toLowerCase().replace(/[^\w\s]/g, '').split(/\s+/);
        words.forEach(word => {
            if (!invertedIndex[word]) {
                invertedIndex[word] = [];
            }
            if (!invertedIndex[word].includes(docID)) {
                invertedIndex[word].push(docID);
            }
        });
    });
    return invertedIndex;
}

const documents = {
    "Article: Pandas Basics": "This article covers the basics of using Pandas in Python.",
    "Tutorial: Data Visualization": "Learn about data visualization techniques with Python libraries.",
    "Case Study: Sales Analysis": "Analyze sales data using Python for a real-world case study."
};

const invertedIndex = createInvertedIndex(documents);
